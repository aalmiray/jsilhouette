58      @Override
59      public void start(Stage primaryStage) throws Exception {
60          primaryStage.setTitle("Image Generator");
61          primaryStage.showingProperty().addListener((observable, oldValue, newValue) -> generateImages(primaryStage));
62          primaryStage.show();
63      }

65      private void generateImages(Stage stage) {
66          Map<String, Pane> nodes = new LinkedHashMap<>();
67          nodes.put("almond", almond(stage));
68          nodes.put("arrow", arrow(stage));
69          nodes.put("asterisk", asterisk(stage));
70          nodes.put("astroid", astroid(stage));
71          nodes.put("cross", cross(stage));
72          nodes.put("donut", donut(stage));
73          nodes.put("lauburu", lauburu(stage));
74          nodes.put("multiround_rectangle", multiround_rectangle(stage));
75          nodes.put("rays", rays(stage));
76          nodes.put("regular_polygon", regular_polygon(stage));
77          nodes.put("roundpin", roundpin(stage));
78          nodes.put("star", star(stage));
79  
80          nodes.forEach((name, grid) -> {
81              Scene scene = new Scene(grid);
82              scene.setFill(Color.WHITE);
83              scene.getStylesheets().add("org/kordamp/jsilhouette/generator.css");
84              stage.setScene(scene);
85  
86              SnapshotParameters params = new SnapshotParameters();
87              params.setFill(Color.TRANSPARENT);
88              WritableImage snapshot = grid.snapshot(params, null);
89  
90              File projectBuildDir = new File(System.getProperty("user.dir"), "build/images");
91              projectBuildDir.mkdirs();
92              try {
93                  ImageIO.write(SwingFXUtils.fromFXImage(snapshot, null), "png", new File(projectBuildDir, "shape_" + name + ".png"));
94              } catch (IOException e) {
95                  e.printStackTrace();
96              }
97          });
98          System.exit(0);
99      }
