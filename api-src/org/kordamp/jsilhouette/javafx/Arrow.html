024 /**
025  * = Arrow
026  *
027  * Defines an arrow shape
028  *
029  * image::shape_arrow.png[]
030  *
031  * == Constraints
032  *
033  *  . 0 +<=+ `depth` +<=+ 1
034  *  . 0 +<=+ `rise` +<=+ 1
035  *
036  * == Style Classes
037  *
038  *  . `silhouette`
039  *  . `silhouette-arrow`
040  *
041  * @author Andres Almiray
042  */
043 public class Arrow extends AbstractSilhouette {
044     private static final Logger LOG = Logger.getLogger(Arrow.class.getName());
045 
046     private DoubleProperty x;
047     private DoubleProperty y;
048     private DoubleProperty width;
049     private DoubleProperty height;
050     private DoubleProperty depth;
051     private DoubleProperty rise;
052 
053     public Arrow() {
054     }
055 
056     public Arrow(double x, double y, double width, double height) {
057         this(x, y, width, height, 0.5, 0.5);
058     }
059 
060     public Arrow(double x, double y, double width, double height, double depth, double rise) {
061         initializing = true;
062         setX(x);
063         setY(y);
064         setWidth(width);
065         setHeight(height);
066         setDepth(depth);
067         setRise(rise);
068         initializing = false;
069         calculateShape();
070     }
071

073         return xProperty().get();
074     }
075

077     public DoubleProperty xProperty() {

171         riseProperty().set(rise);
172     }

174     @Override
175     protected void calculateShape() {
176         double x = getX();
177         double y = getY();
178         double w = getWidth();
179         double h = getHeight();
180         double d = w * validateDepth(getDepth());
181         double r = h * validateRise(getRise()) / 2;
182 
183         Path shape = new PathBuilder()
184             .moveTo(x, y + (h / 2) - r)
185             .lineTo(x + d, y + (h / 2) - r)
186             .lineTo(x + d, y)
187             .lineTo(x + w, y + (h / 2))
188             .lineTo(x + d, y + h)
189             .lineTo(x + d, y + (h / 2) + r)
190             .lineTo(x, y + (h / 2) + r)
191             .build();
192 
193         shape.getStyleClass().addAll("silhouette", "silhoutte-arrow");
194 
195         setShape(shape);
196     }
197 
198     private double validateDepth(double depth) {
199         if (depth < 0 || depth > 1) {
200             LOG.info(() -> "depth (" + depth + ") must be inside the range [0..1]");
201             return 0.5;
202         }
203         return depth;
204     }
205 
206     private double validateRise(double rise) {
207         if (rise < 0 || rise > 1) {
208             LOG.info(() -> "rise (" + rise + ") must be inside the range [0..1]");
209             return 0.5;
210         }
211         return rise;
212     }
