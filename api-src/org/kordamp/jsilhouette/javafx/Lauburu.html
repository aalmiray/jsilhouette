051 package org.kordamp.jsilhouette.javafx;
052 
053 import javafx.beans.property.DoubleProperty;
054 import javafx.beans.property.ObjectProperty;
055 import javafx.beans.property.SimpleDoubleProperty;
056 import javafx.beans.property.SimpleObjectProperty;
057 import javafx.scene.shape.Shape;
058 
059 /**
060  * = Lauburu
061  *
062  * Defines a link:http://en.wikipedia.org/wiki/Lauburu[Lauburu] shape.
063  * 
064  * image::shape_lauburu.png[]
065  * 
066  * == Style Classes
067  * 
068  * . `silhouette`
069  * . `silhouette-lauburu`
070  *
071  * @author Andres Almiray
072  */
073 public class Lauburu extends AbstractCenteredSilhouette {
074     public enum Direction {
075         CLOCKWISE,
076         ANTICLOCKWISE
077     }
078 
079     private DoubleProperty radius;
080     private ObjectProperty<Direction> direction;
081 
082     public Lauburu() {
083     }
084 
085     public Lauburu(double cx, double cy, double radius) {
086         this(cx, cy, radius, Direction.CLOCKWISE);
087     }
088 
089     public Lauburu(double cx, double cy, double radius, Direction direction) {
090         initializing = true;
091         setCenterX(cx);
092         setCenterY(cy);
093         setRadius(radius);
094         setDirection(direction);
095         initializing = false;
096         calculateShape();
097     }
098 
099     public double getRadius() {
100         return radiusProperty().get();
101     }
102 
103     @SuppressWarnings("unchecked")
104     public DoubleProperty radiusProperty() {
105         if (radius == null) {
106             radius = new SimpleDoubleProperty(this, "radius", 0);
107             radius.addListener(updateListener);
108         }
109         return radius;
110     }
111 
112     public void setRadius(double radius) {
113         radiusProperty().set(radius);
114     }
115 
116     public Direction getDirection() {
117         return directionProperty().get();
118     }
119 
120     @SuppressWarnings("unchecked")
121     public ObjectProperty<Direction> directionProperty() {
122         if (direction == null) {
123             direction = new SimpleObjectProperty(this, "direction", Direction.CLOCKWISE);
124             direction.addListener(updateListener);
125         }
126         return direction;
127     }
128 
129     public void setDirection(Direction direction) {
130         directionProperty().set(direction);
131     }
132 
133     @Override
134     protected void calculateShape() {
135         double cx = getCenterX();
136         double cy = getCenterY();
137         double r = getRadius();
138         Direction d = getDirection();
139 
140         Shape shape = new PathBuilder().moveTo(cx, cy)
141             .arcTo(cx + r, cy, r / 2, r / 2, d == Direction.CLOCKWISE)
142             .arcTo(cx + (r / 2), cy, r / 4, r / 4, d == Direction.CLOCKWISE)
143             .arcTo(cx, cy, r / 4, r / 4, d != Direction.CLOCKWISE)
144             .close()
145             .arcTo(cx - r, cy, r / 2, r / 2, d == Direction.CLOCKWISE)
146             .arcTo(cx - (r / 2), cy, r / 4, r / 4, d == Direction.CLOCKWISE)
147             .arcTo(cx, cy, r / 4, r / 4, d != Direction.CLOCKWISE)
148             .close()
149             .arcTo(cx, cy + r, r / 2, r / 2, d == Direction.CLOCKWISE)
150             .arcTo(cx, cy + (r / 2), r / 4, r / 4, d == Direction.CLOCKWISE)
151             .arcTo(cx, cy, r / 4, r / 4, d != Direction.CLOCKWISE)
152             .close()
153             .arcTo(cx, cy - r, r / 2, r / 2, d == Direction.CLOCKWISE)
154             .arcTo(cx, cy - (r / 2), r / 4, r / 4, d == Direction.CLOCKWISE)
155             .arcTo(cx, cy, r / 4, r / 4, d != Direction.CLOCKWISE)
156             .close()
157             .build();
158 
159         shape.getStyleClass().addAll("silhouette", "silhoutte-lauburu");
160 
161         setShape(shape);
162     }
163 }
