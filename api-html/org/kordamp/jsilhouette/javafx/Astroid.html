18 package org.kordamp.jsilhouette.javafx;
19 
20 import javafx.beans.property.DoubleProperty;
21 import javafx.beans.property.SimpleDoubleProperty;
22 import javafx.scene.shape.Path;
23 
24 /**
25  * Astroid
26  * <p>
27  * Defines an <a href="http://en.wikipedia.org/wiki/Astroid">Astroid</a> shape.
28  *
29  * <h3>Style Classes</h3>
30  * <ul>
31  *  <li>silhouette</li>
32  *  <li>silhouette-astroid</li>
33  * </ul>
34  *
35  * @author Andres Almiray
36  */
37 public class Astroid extends AbstractCenteredSilhouette {
38     private DoubleProperty radius;
39 
40     public Astroid() {
41     }
42 
43     public Astroid(double cx, double cy, double radius) {
44         initializing = true;
45         setCenterX(cx);
46         setCenterY(cy);
47         setRadius(radius);
48         initializing = false;
49         calculateShape();
50     }
51 
52     public double getRadius() {
53         return radiusProperty().get();
54     }
55 
56     @SuppressWarnings("unchecked")
57     public DoubleProperty radiusProperty() {
58         if (radius == null) {
59             radius = new SimpleDoubleProperty(this, "radius", 0);
60             radius.addListener(updateListener);
61         }
62         return radius;
63     }
64 
65     public void setRadius(double radius) {
66         radiusProperty().set(radius);
67     }
68 
69     @Override
70     protected void calculateShape() {
71         PathBuilder p = new PathBuilder();
72 
73         double cx = getCenterX();
74         double cy = getCenterY();
75         double r = getRadius();
76 
77         Path path = p.moveTo(cx - r, cy)
78             .arcTo(cx, cy - r, r, r, false)
79             .arcTo(cx + r, cy, r, r, false)
80             .arcTo(cx, cy + r, r, r, false)
81             .arcTo(cx - r, cy, r, r, false)
82             .build();
83 
84         path.getStyleClass().addAll("silhouette", "silhoutte-astroid");
85 
86         setShape(path);
87     }
88 }
